#include<iostream>
#include<stack>
using namespace std;

int main() {

    stack<int> stack;
    stack.push(21);
    stack.push(22);
    stack.push(24);
    stack.push(25);

    cout << "Size: " << stack.size() << endl;
    cout << "Top: " << stack.top() << endl;
    stack.pop();
    stack.push(70);
    stack.push(80);
    stack.pop();
    cout << "Top: " << stack.top() << endl;
    cout << "Empty(1) or not(0): " << stack.empty() << endl << endl;

    while (!stack.empty()) {
        cout << stack.top() << " ";
        stack.pop();
    }

    return 0
}
